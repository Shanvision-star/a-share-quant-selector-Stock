# 进入你的项目根目录
cd /home/Opendingding/a-share-quant-selector-main

# 1. 新建docs文件夹（专门存放文档、手册、个人笔记）
mkdir -p docs

# 2. 一键生成带注释的手册文件到docs文件夹
cat > docs/git_操作手册.txt << 'EOF'
# Git 核心操作指令手册（量化项目专用）
# 适用场景：A股量化选股项目代码版本管理、历史回溯、版本回退
# 项目路径：/home/Opendingding/a-share-quant-selector-main

# ================ 一、首次初始化配置（仅第一次使用执行1次） ================
# 1. 进入你的项目根目录（所有Git操作必须先进入这个目录，否则无效）
cd /home/Opendingding/a-share-quant-selector-main

# 2. 初始化Git仓库：把当前文件夹变成Git可管理的版本控制项目
git init

# 3. 配置提交者用户名（随便写，用来标记是谁提交的代码）
git config user.name "你的名字"

# 4. 配置提交者邮箱（随便写，和用户名配套即可）
git config user.email "你的邮箱@example.com"

# 5. 提交项目初始版本：第一次把所有代码保存成第一个可回溯的版本
git add .                                  # 把当前文件夹所有文件加入暂存区
git commit -m "初始版本：量化选股项目完整代码"  # 保存版本，必须写清备注


# ================ 二、日常核心操作（每次改完代码必执行，最常用） ================
# 【标准3步走，10秒完成代码保存】
# 1. 把所有修改过的代码、新增的文件加入暂存区（. 代表当前目录所有文件）
git add .

# 2. 保存成一个新版本，【必须写清备注！！】建议格式：时间+修改内容
# 正确示例：git commit -m "2026-03-28 18:00 修复钉钉通知发送失败bug"
# 正确示例：git commit -m "2026-03-28 18:30 优化选股策略，新增均线过滤条件"
git commit -m "你的版本备注"

# 3. 【可选】如果绑定了GitHub/Gitee远程仓库，把本地版本推送到云端备份
git push origin main


# ================ 三、查看历史版本（找几分钟/几小时前的代码，核心需求） ================
# 1. 极简查看所有历史版本（最常用，一眼找到目标版本号和备注）
# 输出格式：版本号前7位 + 你写的版本备注
git log --oneline

# 2. 查看某个版本的详细修改内容（先看改了什么，再决定要不要回退）
# 格式：git show 版本号（复制git log --oneline里的前7位即可）
# 示例：git show a1b2c3d
git show 你的目标版本号

# 3. 查看某个特定文件的历史修改记录（比如只想看main.py的改动）
# 示例：git log --oneline main.py
git log --oneline 你要查的文件名

# 4. 查看某个文件在旧版本里的完整内容（不用回退，直接预览旧代码）
# 示例：git show a1b2c3d:main.py
git show 版本号:文件名


# ================ 四、版本回退/恢复（改坏代码救急，分2种安全级别） ================
# ---------- 【推荐：安全模式 软回退】----------
# 效果：回到目标版本，保留你现在写的所有代码，只是撤销了commit提交
# 适用场景：刚才提交错了，想重新修改再提交，不会丢失任何代码
# 示例：git reset --soft d4e5f6g
git reset --soft 你要回退的目标版本号

# ---------- 【慎用：强制模式 硬回退】----------
# ⚠️ 警告：执行后，目标版本之后写的所有代码会被彻底删除，无法恢复！！
# 效果：彻底回到目标版本的状态，所有后续修改全部清空
# 适用场景：确定现在的代码全是错的，要彻底回到之前的正常版本
# 示例：git reset --hard d4e5f6g
git reset --hard 你要回退的目标版本号

# ---------- 【单文件恢复】----------
# 效果：只把某一个文件恢复到目标版本，其他所有文件完全不变
# 适用场景：只有一个文件改坏了，其他文件都是好的，不用全量回退
# 示例：git checkout a1b2c3d -- main.py
git checkout 目标版本号 -- 要恢复的文件名


# ================ 五、远程仓库备份（可选，云端备份代码防丢失） ================
# 1. 绑定远程仓库（在GitHub/Gitee创建空仓库后，复制仓库地址执行）
# 示例：git remote add origin https://gitee.com/你的账号/你的仓库名.git
git remote add origin 你的远程仓库地址

# 2. 把本地代码推送到远程仓库（第一次推送用）
git push -u origin main

# 3. 拉取远程仓库的最新代码（多人协作/换电脑时用）
git pull origin main
EOF

# 3. 把个人手册加入.gitignore，避免被Git提交污染仓库记录
echo "docs/*.txt" >> .gitignore
